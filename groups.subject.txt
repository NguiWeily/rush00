Based on the instructions given, here's an example implementation of the `main.c`, `ft_putchar.c`, and `rush00.c` files for the Rush00 exercise:

**main.c**
```c
int main()
{
    rush(5, 5);
    return 0;
}
```

**ft_putchar.c**
```c
#include <unistd.h>

void ft_putchar(char c)
{
    write(1, &c, 1);
}
```

**rush00.c**
```c
void ft_putchar(char c);

void print_line(int width, char first, char middle, char last)
{
    int i;

    ft_putchar(first);
    i = 1;
    while (i < width - 1)
    {
        ft_putchar(middle);
        i++;
    }
    if (width > 1)
        ft_putchar(last);
    ft_putchar('\n');
}

void rush(int x, int y)
{
    int i;

    if (x <= 0 || y <= 0)
        return;
    
    print_line(x, 'o', '-', 'o');
    i = 1;
    while (i < y - 1)
    {
        print_line(x, '|', ' ', '|');
        i++;
    }
    if (y > 1)
        print_line(x, 'o', '-', 'o');
}
```
